'use client'
import { useEffect, useState } from 'react'

interface Settings {
  stripe_publishable_key: string
  stripe_secret_key: string
  stripe_webhook_secret: string
  canadapost_api_key: string
  canadapost_customer_number: string
  shipping_markup_percent: string
  email_from: string
  email_smtp_host: string
  email_smtp_port: string
}

export default function AdminSettings() {
  const [settings, setSettings] = useState<Settings>({
    stripe_publishable_key: '',
    stripe_secret_key: '',
    stripe_webhook_secret: '',
    canadapost_api_key: '',
    canadapost_customer_number: '',
    shipping_markup_percent: '0',
    email_from: 'hello@danditos.nex.monster',
    email_smtp_host: 'premium340.web-hosting.com',
    email_smtp_port: '465'
  })
  const [saving, setSaving] = useState(false)
  const [saved, setSaved] = useState(false)

  useEffect(() => {
    fetch('/api/settings')
      .then(res => res.json())
      .then(data => {
        setSettings(prev => ({ ...prev, ...data }))
      })
      .catch(console.error)
  }, [])

  const handleSave = async (key: string, value: string) => {
    setSaving(true)
    await fetch('/api/settings', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ key, value })
    })
    setSaving(false)
    setSaved(true)
    setTimeout(() => setSaved(false), 2000)
  }

  const handleChange = (key: keyof Settings, value: string) => {
    setSettings(prev => ({ ...prev, [key]: value }))
  }

  return (
    <div>
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '32px' }}>
        <h1>Settings</h1>
        {saved && <span style={{ color: '#27ae60' }}>Settings saved!</span>}
      </div>

      <div className="checkout-form" style={{ marginBottom: '24px' }}>
        <h3>Stripe Configuration</h3>
        
        <div className="form-group">
          <label>Publishable Key</label>
          <input 
            type="text" 
            value={settings.stripe_publishable_key}
            onChange={(e) => handleChange('stripe_publishable_key', e.target.value)}
            onBlur={(e) => handleSave('stripe_publishable_key', e.target.value)}
            placeholder="pk_live_..."
          />
        </div>

        <div className="form-group">
          <label>Secret Key</label>
          <input 
            type="password" 
            value={settings.stripe_secret_key}
            onChange={(e) => handleChange('stripe_secret_key', e.target.value)}
            onBlur={(e) => handleSave('stripe_secret_key', e.target.value)}
            placeholder="sk_live_..."
          />
        </div>

        <div className="form-group">
          <label>Webhook Secret</label>
          <input 
            type="password" 
            value={settings.stripe_webhook_secret}
            onChange={(e) => handleChange('stripe_webhook_secret', e.target.value)}
            onBlur={(e) => handleSave('stripe_webhook_secret', e.target.value)}
            placeholder="whsec_..."
          />
        </div>
      </div>

      <div className="checkout-form" style={{ marginBottom: '24px' }}>
        <h3>Canada Post Configuration</h3>
        
        <div className="form-group">
          <label>API Key</label>
          <input 
            type="password" 
            value={settings.canadapost_api_key}
            onChange={(e) => handleChange('canadapost_api_key', e.target.value)}
            onBlur={(e) => handleSave('canadapost_api_key', e.target.value)}
            placeholder="Your Canada Post API key"
          />
        </div>

        <div className="form-group">
          <label>Customer Number</label>
          <input 
            type="text" 
            value={settings.canadapost_customer_number}
            onChange={(e) => handleChange('canadapost_customer_number', e.target.value)}
            onBlur={(e) => handleSave('canadapost_customer_number', e.target.value)}
            placeholder="Your Canada Post customer number"
          />
        </div>

        <div className="form-group">
          <label>Shipping Markup (%)</label>
          <input 
            type="number" 
            step="0.1"
            value={settings.shipping_markup_percent}
            onChange={(e) => handleChange('shipping_markup_percent', e.target.value)}
            onBlur={(e) => handleSave('shipping_markup_percent', e.target.value)}
            placeholder="0"
          />
          <small style={{ color: '#888', marginTop: '4px', display: 'block' }}>
            Percentage added to Canada Post rates (e.g., 20 = 20% markup)
          </small>
        </div>
      </div>

      <div className="checkout-form">
        <h3>Email Configuration</h3>
        
        <div className="form-group">
          <label>From Address</label>
          <input 
            type="email" 
            value={settings.email_from}
            onChange={(e) => handleChange('email_from', e.target.value)}
            onBlur={(e) => handleSave('email_from', e.target.value)}
            placeholder="hello@danditos.nex.monster"
          />
        </div>

        <div className="form-row">
          <div className="form-group">
            <label>SMTP Host</label>
            <input 
              type="text" 
              value={settings.email_smtp_host}
              onChange={(e) => handleChange('email_smtp_host', e.target.value)}
              onBlur={(e) => handleSave('email_smtp_host', e.target.value)}
              placeholder="premium340.web-hosting.com"
            />
          </div>
          <div className="form-group">
            <label>SMTP Port</label>
            <input 
              type="text" 
              value={settings.email_smtp_port}
              onChange={(e) => handleChange('email_smtp_port', e.target.value)}
              onBlur={(e) => handleSave('email_smtp_port', e.target.value)}
              placeholder="465"
            />
          </div>
        </div>
      </div>
    </div>
  )
}
