'use client'
import { useEffect, useState } from 'react'

interface Product {
  id: number
  name: string
  slug: string
  description: string
  heat_level: string
  price: string
  stock: number
  image_url: string
  is_active: boolean
}

export default function AdminProducts() {
  const [products, setProducts] = useState<Product[]>([])
  const [editingProduct, setEditingProduct] = useState<Product | null>(null)
  const [isNew, setIsNew] = useState(false)
  const [form, setForm] = useState({
    name: '',
    slug: '',
    description: '',
    heat_level: 'medium',
    price: '',
    stock: '',
    image_url: '',
    is_active: true
  })
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    loadProducts()
  }, [])

  const loadProducts = () => {
    fetch('/api/products')
      .then(res => res.json())
      .then(data => {
        setProducts(data)
        setLoading(false)
      })
      .catch(() => setLoading(false))
  }

  const handleEdit = (product: Product) => {
    setEditingProduct(product)
    setForm({
      name: product.name,
      slug: product.slug,
      description: product.description || '',
      heat_level: product.heat_level,
      price: product.price,
      stock: product.stock.toString(),
      image_url: product.image_url,
      is_active: product.is_active
    })
    setIsNew(false)
  }

  const handleNew = () => {
    setEditingProduct(null)
    setForm({
      name: '',
      slug: '',
      description: '',
      heat_level: 'medium',
      price: '',
      stock: '',
      image_url: '',
      is_active: true
    })
    setIsNew(true)
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    
    const url = isNew ? '/api/admin/products' : `/api/admin/products?id=${editingProduct?.id}`
    const method = isNew ? 'POST' : 'PUT'

    const response = await fetch(url, {
      method,
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(form)
    })

    if (response.ok) {
      loadProducts()
      setEditingProduct(null)
      setIsNew(false)
    }
  }

  const handleDelete = async (id: number) => {
    if (!confirm('Are you sure you want to delete this product?')) return
    
    await fetch(`/api/admin/products?id=${id}`, { method: 'DELETE' })
    loadProducts()
  }

  if (loading) {
    return <p>Loading...</p>
  }

  return (
    <div>
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '32px' }}>
        <h1>Products</h1>
        <button onClick={handleNew} className="btn btn-primary">
          + Add Product
        </button>
      </div>

      {editingProduct || isNew ? (
        <form onSubmit={handleSubmit} className="admin-form">
          <h3 style={{ marginBottom: '24px' }}>{isNew ? 'New Product' : 'Edit Product'}</h3>
          
          <div className="form-group">
            <label>Name</label>
            <input 
              type="text" 
              value={form.name}
              onChange={(e) => setForm({...form, name: e.target.value})}
              required
            />
          </div>

          <div className="form-group">
            <label>Slug</label>
            <input 
              type="text" 
              value={form.slug}
              onChange={(e) => setForm({...form, slug: e.target.value.toLowerCase().replace(/[^a-z0-9-]/g, '-')})}
              required
            />
          </div>

          <div className="form-group">
            <label>Description</label>
            <textarea 
              value={form.description}
              onChange={(e) => setForm({...form, description: e.target.value})}
            />
          </div>

          <div className="form-row">
            <div className="form-group">
              <label>Heat Level</label>
              <select value={form.heat_level} onChange={(e) => setForm({...form, heat_level: e.target.value})}>
                <option value="mild">Mild</option>
                <option value="medium">Medium</option>
                <option value="hot">Hot</option>
                <option value="extreme">Extreme</option>
              </select>
            </div>
            <div className="form-group">
              <label>Price ($)</label>
              <input 
                type="number" 
                step="0.01"
                value={form.price}
                onChange={(e) => setForm({...form, price: e.target.value})}
                required
              />
            </div>
          </div>

          <div className="form-row">
            <div className="form-group">
              <label>Stock</label>
              <input 
                type="number" 
                value={form.stock}
                onChange={(e) => setForm({...form, stock: e.target.value})}
                required
              />
            </div>
            <div className="form-group">
              <label>Active</label>
              <select value={form.is_active ? 'true' : 'false'} onChange={(e) => setForm({...form, is_active: e.target.value === 'true'})}>
                <option value="true">Yes</option>
                <option value="false">No</option>
              </select>
            </div>
          </div>

          <div className="form-group">
            <label>Image URL</label>
            <input 
              type="url" 
              value={form.image_url}
              onChange={(e) => setForm({...form, image_url: e.target.value})}
            />
          </div>

          <div style={{ display: 'flex', gap: '16px' }}>
            <button type="submit" className="btn btn-primary">
              {isNew ? 'Create Product' : 'Save Changes'}
            </button>
            <button type="button" onClick={() => { setEditingProduct(null); setIsNew(false) }} className="btn btn-secondary">
              Cancel
            </button>
          </div>
        </form>
      ) : (
        <table className="admin-table">
          <thead>
            <tr>
              <th>Name</th>
              <th>Heat</th>
              <th>Price</th>
              <th>Stock</th>
              <th>Active</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>
            {products.map(product => (
              <tr key={product.id}>
                <td>{product.name}</td>
                <td><span className={`heat-badge heat-${product.heat_level}`}>{product.heat_level}</span></td>
                <td>${parseFloat(product.price).toFixed(2)}</td>
                <td>{product.stock}</td>
                <td>{product.is_active ? '✓' : '✗'}</td>
                <td>
                  <button onClick={() => handleEdit(product)} className="btn btn-secondary" style={{ padding: '6px 16px', marginRight: '8px' }}>
                    Edit
                  </button>
                  <button onClick={() => handleDelete(product.id)} style={{ background: 'var(--fire-red)', border: 'none', color: 'white', padding: '6px 16px', borderRadius: '4px', cursor: 'pointer' }}>
                    Delete
                  </button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  )
}
